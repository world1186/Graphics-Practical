#include<stdio.h>
#include<graphics.h>
#include<conio.h>
#include<dos.h>
int main()
{

    int x,y,x1=100,y1=150,x2=200,y2=300,p,dx,dy;

        printf("ANITA KARKI");
        printf("Enter the x_coordinate of first point");
        scanf("%d",&x1);
        printf("Enter the y_coordinate of first point");
        scanf("%d",&y1);
        printf("Enter the x_coordinate of second point");
        scanf("%d",&x2);
        printf("Enter the y_coordinate of second point");
        scanf("%d",&y2);
        x=x1;
        y=y1;
        dx=x2-x1;
        dy=y2-y1;
        int gd=DETECT,gm;
        initgraph(&gd,&gm,"");

        putpixel(x,y,2);

        p=(2*dy-dx);
        while(x>=2)
        {

            if(p<0)
            {

                x=x+1;
                p=p+2*dy;
            }
            else{
    x=x+1;
    y=y+1;
    p=p+(2*dy)-(2*dx);
            }
            putpixel(x,y,7);
        }
        getch();
        closegraph();
}
